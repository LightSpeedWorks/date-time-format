date-time-format.js
====

# USAGE:

```js
var DateTime = require('date-time-format');
var toDateTimeString = DateTime.toDateTimeString;
var toDateString     = DateTime.toDateString;
var toTimeString     = DateTime.toTimeString;

DateTime.extendDateToDateTimeString();
DateTime.extendDateToString();

var toDateTimeString = require('date-time-format').toDateTimeString;
var toDateString = require('date-time-format').toDateString;
var toTimeString = require('date-time-format').toTimeString;

require('date-time-format').extendDateToDateTimeString();
require('date-time-format').extendDateToString();
```
